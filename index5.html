/** * 處理 POST 請求：接收來自 HTML 表單的資料並寫入指定的試算表分頁 */ function doPost(e) { try { const lock = LockService.getScriptLock(); // 嘗試取得鎖定，避免多個使用者同時送出導致資料錯亂（等待 10 秒） lock.waitLock(10000); // 取得當前試算表 const ss = SpreadsheetApp.getActiveSpreadsheet(); // 根據您的需求：指定工作頁籤名稱為 'breakfast' const sheet = ss.getSheetByName('breakfast'); // 防錯處理：如果找不到該名稱的頁籤，則傳回錯誤 if (!sheet) { throw new Error("找不到名為 'breakfast' 的工作頁籤，請檢查名稱是否正確。"); } // 取得第一列的標題（Header），用來比對傳入的 JSON 欄位 const lastColumn = sheet.getLastColumn(); if (lastColumn === 0) { throw new Error("工作表第一列必須先手動填入標題欄位（例如：時間、姓名 / 桌號、原味蛋餅...）"); } const headers = sheet.getRange(1, 1, 1, lastColumn).getValues()[0]; // 解析前端傳來的 JSON 資料 const data = JSON.parse(e.postData.contents); // 依照標題順序重組資料行，並自動加入目前的伺服器時間 const newRow = headers.map(function(header) { if (header === "時間") return new Date(); // 如果有時間欄位則自動填入 return data[header] !== undefined ? data[header] : ""; }); // 將新的一行寫入試算表 sheet.appendRow(newRow); // 回傳成功訊息給前端 return ContentService .createTextOutput(JSON.stringify({ "result": "success", "row": sheet.getLastRow() })) .setMimeType(ContentService.MimeType.JSON); } catch (error) { // 發生錯誤時回傳錯誤訊息 return ContentService .createTextOutput(JSON.stringify({ "result": "error", "error": error.toString() })) .setMimeType(ContentService.MimeType.JSON); } finally { // 釋放鎖定 lock.releaseLock(); } } /** * 處理 GET 請求：用來檢查 API 是否部署成功 */ function doGet(e) { return ContentService.createTextOutput("早餐店 API 運作中。目標頁籤：'breakfast'。"); }
